\section{Schr\"odinger's wave equation}

\begin{align*}
	\frac{-\hbar^2}{2m} \frac{d^2}{dx^2} \Psi(x) + V(x) &= E\Psi(x)\\
	\hat{H}\Psi(x) &= E\Psi(x) \quad \textrm{where, $\hat{H}$ is the Hamiltonian Operator}
\end{align*}
